        """Test diff parsing functionality."""
        parser = DiffParser()
        diff = """
        diff --git a/file.py b/file.py
        index abc..def 100644
        --- a/file.py
        +++ b/file.py
        @@ -1,3 +1,4 @@
        """
        chunks = parser.parse(diff)
        assert len(chunks) == 1
