### 5.2 CI/CD

<CodeGroup>
  <CodeBlock title="GitHub Actions">
    ```yaml
    name: Release
    on:
      push:
        tags: ['v*']
    
    jobs:
      release:
        runs-on: ubuntu-latest
        steps:
          - uses: actions/checkout@v3
          - uses: actions/setup-python@v4
          
          - name: Build and publish
            run: |
              python -m build
              python -m twine upload dist/*
    ```
  </CodeBlock>
</CodeGroup>
